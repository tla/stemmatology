48=head2 __autovivify
49
50Try to see if the type constraint has yet been defined and if so create it.
51
52=cut

54sub __autovivify {
55 my ($self) = @_;
56 if(my $tc = $self->{instance}) {
57 return $tc;
58 } elsif( my $new_tc = Moose::Util::TypeConstraints::find_type_constraint($self->name)) {
59 $self->{instance} = $new_tc;
60 return $new_tc;
61 } else {
62 return;
63 }
64}
