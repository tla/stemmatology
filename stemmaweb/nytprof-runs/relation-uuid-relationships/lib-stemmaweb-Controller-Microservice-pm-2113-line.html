226sub _parse_to_tradition {
227 my $req = shift;
228 my $type = $req->body_params->{'type'};
229 my $name = $req->param('name') || 'Collation graph';
230 my $data = $req->body_params->{'data'};

236 $opts->{'sep_char'} = ',' if $type eq 'CSV';
