676 # TODO this needs overrides in a number of subclasses,
677 # in particular in Bio::Phylo::Taxa and Bio::Phylo::Taxa::Taxon
678 # classes because of the asymmetry between set_forest/get_forests,
679 # set_node/get_nodes etc.
680 sub clone {
681 my ( $self, %subs ) = @_;
682
683 # may not work yet! warn user
684 $logger->info("cloning is experimental, use with caution");
685
686 # get inheritance tree
687 my ( $class, $isa, $seen ) = ( ref($self), [], {} );
688 _recurse_isa( $class, $isa, $seen );
689
690 # walk symbol table, get symmetrical set_foo/get_foo pairs
691 my %methods;
692 for my $package ( $class, @{$isa} ) {
693 my %symtable;
694 eval "\%symtable = \%${package}::";
695 SETTER: for my $setter ( keys %symtable ) {
696 next SETTER if $setter !~ m/^set_/;
697 my $getter = $setter;
698 $getter =~ s/^s/g/;
699 next SETTER if not exists $symtable{$getter};
700
701 # have a symmetrical set_foo/get_foo pair, check
702 # if they're code (not variables, for example)
703 my $get_ref = $class->can($getter);
704 my $set_ref = $class->can($setter);
705 if ( looks_like_instance( $get_ref, 'CODE' )
706 and looks_like_instance( $set_ref, 'CODE' ) )
707 {
708 $methods{$getter} = $setter;
709 }
710 }
711 }
712
713 # instantiate the clone
714 my @new;
715 if ( $subs{'new'} ) {
716 @new = @{ $subs{'new'} };
717 delete $subs{'new'};
718 }
719 my $clone = $class->new(@new);
720
721 # populate the clone
722 for my $getter ( keys %methods ) {
723 my $setter = $methods{$getter};
724 if ( exists $subs{$setter} ) {
725 $logger->info("method $setter for $clone overridden");
726 if ( looks_like_instance( $subs{$setter}, 'CODE' ) ) {
727 $subs{$setter}->( $self, $clone );
728 }
729 delete $subs{$setter};
730 }
731 else {
732 eval {
733 $logger->info("copying $getter => $setter");
734 my $value = $self->$getter;
735 if ( defined $value ) {
736 $clone->$setter($value);
737 }
738 };
739 if ($@) {
740 $logger->warn("failed copy of $getter => $setter: \n$@");
741 }
742 }
743 }
744
745 # execute additional code refs
746 $_->( $self, $clone )
747 for ( grep { looks_like_instance( $_, 'CODE' ) } values %subs );
748 return $clone;
749 }
