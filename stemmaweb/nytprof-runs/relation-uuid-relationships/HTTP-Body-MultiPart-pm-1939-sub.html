172 my $self = shift;

286 }

288 if ( $part->{fh} && ( my $length = length( $part->{data} ) ) ) {
289 $part->{fh}->write( substr( $part->{data}, 0, $length, '' ), $length );
290 }
291
292 if ( $part->{done} ) {
293

295 if ( $part->{filename} ne "" ) {
296 $part->{fh}->close if defined $part->{fh};
297
298 delete @{$part}{qw[ data done fh ]};
