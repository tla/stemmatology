215 sub get_link {
216 my $self = shift;
217 my %opt = @_;
218 my $id = $opt{'-source'}->get_id;
219
220 # have to get many objects,
221 # i.e. source was a taxon/taxa
222 if ( defined $opt{'-type'} ) {
223 my $type = $opt{'-type'};
224 my @ids = grep { $one_to_many{$id}->{$_} == $type } keys %{ $one_to_many{$id} };
225 my @result = @object[@ids];
226 return \@result;
227 }
228
229 # have to get one object, i.e. source
230 # was something that links to taxon/taxa
231 else {
232 return exists $one_to_one{$id} ? $object[$one_to_one{$id}] : undef;
233 }
